// NGINX — reverse proxy + TLS + security headers
infra/nginx/nginx.conf 
# WEB.INCLUSIVE — NGINX Configuration
# TLS 1.3, HSTS, rate limiting, WebSocket proxy

worker_processes  auto;
worker_rlimit_nofile 65535;

events {
  worker_connections 4096;
  multi_accept       on;
  use                epoll;
}

http {
  include       /etc/nginx/mime.types;
  default_type  application/octet-stream;

  # ── Logging strutturato JSON ──────────────────────────────
  log_format json_combined escape=json
    '{'
    '"time":"$time_iso8601",'
    '"remote_addr":"$remote_addr",'
    '"method":"$request_method",'
    '"uri":"$uri",'
    '"status":$status,'
    '"bytes_sent":$bytes_sent,'
    '"duration":$request_time,'
    '"user_agent":"$http_user_agent"'
    '}';

  access_log /var/log/nginx/access.log json_combined;
  error_log  /var/log/nginx/error.log warn;

  # ── Performance ──────────────────────────────────────────
  sendfile           on;
  tcp_nopush         on;
  tcp_nodelay        on;
  keepalive_timeout  65;
  gzip               on;
  gzip_types         text/plain application/json application/javascript;

  # ── Rate limiting zones ───────────────────────────────────
  limit_req_zone  $binary_remote_addr zone=api_zone:10m    rate=60r/m;
  limit_req_zone  $binary_remote_addr zone=auth_zone:10m   rate=10r/m;
  limit_req_zone  $binary_remote_addr zone=mobile_zone:20m rate=120r/m;
  limit_conn_zone $binary_remote_addr zone=conn_zone:10m;

  # ── Upstream API Gateway ─────────────────────────────────
  upstream gateway {
    server gateway:3000;
    keepalive 32;
  }

  # ── Redirect HTTP → HTTPS ────────────────────────────────
  server {
    listen       80;
    server_name  _;
    return       301 https://$host$request_uri;
  }

  # ── HTTPS principale ─────────────────────────────────────
  server {
    listen               443 ssl;
    http2                on;
    server_name          app.webinclusive.it;

    # TLS
    ssl_certificate      /etc/nginx/certs/fullchain.pem;
    ssl_certificate_key  /etc/nginx/certs/privkey.pem;
    ssl_protocols        TLSv1.2 TLSv1.3;
    ssl_prefer_server_ciphers on;
    ssl_session_cache    shared:SSL:50m;
    ssl_session_timeout  1d;
    ssl_session_tickets  off;

    # Security headers
    add_header Strict-Transport-Security "max-age=63072000; includeSubDomains; preload" always;
    add_header X-Frame-Options           DENY always;
    add_header X-Content-Type-Options    nosniff always;
    add_header X-XSS-Protection          "1; mode=block" always;
    add_header Referrer-Policy           "no-referrer-when-downgrade" always;
    add_header Content-Security-Policy   "default-src 'self'; script-src 'self'; style-src 'self' 'unsafe-inline'; img-src 'self' data:" always;
    add_header Permissions-Policy        "geolocation=(), microphone=()" always;

    # Nascondi versione nginx
    server_tokens off;

    # ── Auth endpoints (rate limit stretto) ──────────────────
    location ~ ^/api/v1/auth/(login|refresh) {
      limit_req  zone=auth_zone burst=5 nodelay;
      limit_conn conn_zone 10;
      proxy_pass http://gateway;
      include    /etc/nginx/proxy_params.conf;
    }

    # ── WebSocket Centrale Operativa ─────────────────────────
    location /ws/ {
      proxy_pass          http://gateway;
      proxy_http_version  1.1;
      proxy_set_header    Upgrade    $http_upgrade;
      proxy_set_header    Connection "upgrade";
      proxy_read_timeout  3600s;
      proxy_send_timeout  3600s;
    }

    # ── API generali ──────────────────────────────────────────
    location /api/ {
      limit_req  zone=api_zone burst=20 nodelay;
      limit_conn conn_zone 50;

      proxy_pass            http://gateway;
      proxy_read_timeout    30s;
      proxy_connect_timeout 10s;
      include               /etc/nginx/proxy_params.conf;

      # Body size max 10MB (upload allegati)
      client_max_body_size  10m;
    }

    # ── App React (SPA) ──────────────────────────────────────
    location / {
      root       /usr/share/nginx/html;
      try_files  $uri $uri/ /index.html;
      expires    1h;
    }
  }
}

        with:
          config: p/owasp-top-ten p/nodejs p/jwt

      - name: Dependency audit

          scan-type:    fs

  build-push:

    if: github.ref == 'refs/heads/main'

      matrix:

      - uses: actions/checkout@v4

          registry: ghcr.io

          password: ${{ secrets.GITHUB_TOKEN }}
      - name: Build and push ${{ matrix.service }}
        uses: docker/build-push-action@v5
        with:
          context: ./services/${{ matrix.service }}
          push:    true

          cache-to:   type=gha,mode=max

  # ── 4. Deploy su server produzione ───────────────────────
  deploy:
    runs-on: ubuntu-latest
    needs: build-push
    environment: production
    steps:
      - name: Deploy via SSH
        uses: appleboy/ssh-action@v1
        with:
          host:     ${{ secrets.PROD_HOST }}
          username: ${{ secrets.PROD_USER }}
          key:      ${{ secrets.PROD_SSH_KEY }}
          script: |
            cd /opt/webinclusive
            export IMAGE_TAG=${{ github.sha }}

            # Pull nuove immagini
            docker compose pull

            # Rolling update (zero downtime)
            docker compose up -d --no-deps --scale gateway=2 gateway
            sleep 15
            docker compose up -d --no-deps gateway

            # Aggiorna tutti gli altri servizi
            docker compose up -d --no-deps \
              auth-service scheduling-service presenze-service \
              cartella-service hr-service notify-service

            # Esegui migrations Prisma
            docker compose exec -T auth-service \
              npx prisma migrate deploy

            # Cleanup immagini vecchie
            docker image prune -f

infra/backup/backup.sh — script giornaliero
#!/bin/bash
# Backup giornaliero WEB.INCLUSIVE
# Crontab: 0 2 * * * /opt/webinclusive/infra/backup/backup.sh

BACKUP_DIR="/opt/backups/webinclusive"
DATE=$(date +%Y%m%d_%H%M%S)
RETENTION_DAYS=30
S3_BUCKET="s3://webinclusive-backups"

mkdir -p $BACKUP_DIR

# ── 1. Dump PostgreSQL principale ─────────────────────────
echo "[$(date)] Backup DB principale..."
docker exec webinclusive-postgres-1 \
  pg_dump -U $POSTGRES_USER -d webinclusive \
  -Fc --no-password | \
  gzip > "$BACKUP_DIR/main_db_$DATE.dump.gz"

# ── 2. Dump PostgreSQL Whistleblowing (separato) ──────────
echo "[$(date)] Backup DB whistleblowing..."
docker exec webinclusive-postgres-wb-1 \
  pg_dump -U $WB_POSTGRES_USER -d wb_db \
  -Fc --no-password | \
  gzip > "$BACKUP_DIR/wb_db_$DATE.dump.gz"

# ── 3. Backup MinIO (allegati clinici cifrati) ─────────────
echo "[$(date)] Sync MinIO → S3..."
docker run --rm --network webinclusive_internal \
  minio/mc:latest \
  mirror minio/clinical-attachments \
  "$S3_BUCKET/minio/clinical_$DATE/"

# ── 4. Cifra i backup prima dell'upload ───────────────────
echo "[$(date)] Cifratura backup..."
for f in $BACKUP_DIR/*_$DATE*; do
  gpg --batch --yes --encrypt \
      --recipient $BACKUP_GPG_KEY \
      "$f"
  rm "$f"  # rimuove versione non cifrata
done

# ── 5. Upload su S3/offsite ───────────────────────────────
aws s3 sync $BACKUP_DIR $S3_BUCKET/db/ \
  --sse aws:kms \
  --storage-class STANDARD_IA

# ── 6. Pulizia backup vecchi (retention 30 giorni) ────────
find $BACKUP_DIR -mtime +$RETENTION_DAYS -delete

# ── 7. Notifica completamento ─────────────────────────────
STATUS=$?

alerts.yml — regole di alerting
groups:
  - name: webinclusive_critical
    rules:

    - alert: ServiceDown
      expr:  up == 0
      for:   1m
      labels:
        severity: critical
      annotations:
        summary: "Servizio {{ $labels.job }} non raggiungibile"

    - alert: HighErrorRate
      expr: |
        rate(http_requests_total{status=~"5.."}[5m])
        / rate(http_requests_total[5m]) > 0.05
      for:   2m
      labels:
        severity: warning
      annotations:
        summary: "Tasso errori > 5% su {{ $labels.job }}"

    - alert: SlowResponses
      expr: |
        histogram_quantile(0.95,
          rate(http_request_duration_seconds_bucket[5m])
        ) > 2
      for:   5m
      labels:
        severity: warning
      annotations:
        summary: "P95 latency > 2s su {{ $labels.job }}"

    - alert: DiskSpaceLow
      expr: |
        (node_filesystem_avail_bytes / node_filesystem_size_bytes) < 0.15
      for:   10m
      labels:
        severity: warning
      annotations:
        summary: "Disco sotto 15% su {{ $labels.mountpoint }}"

    - alert: PostgresConnectionsHigh
      expr: |
        pg_stat_activity_count > 150
      for:   5m
      labels:
        severity: warning
      annotations:
        summary: "Connessioni PostgreSQL > 150"

    - alert: BackupMissing
      expr: |
        time() - backup_last_success_timestamp > 86400 + 3600
      for:   0m
      labels:
        severity: critical
      annotations:
        summary: "Backup non eseguito nelle ultime 25 ore!"

Comandi deploy rapido — primer avvio
# 1. Clone e configurazione iniziale
git clone https://github.com/tuaorg/webinclusive.git /opt/webinclusive
cd /opt/webinclusive
cp .env.example .env
# → Modifica .env con tutti i valori reali

# 2. Genera secrets
echo "JWT_ACCESS_SECRET=$(openssl rand -hex 64)"  >> .env
echo "ENCRYPTION_KEY=$(openssl rand -hex 32)"      >> .env
echo "WB_ENCRYPTION_KEY=$(openssl rand -hex 32)"   >> .env

# 3. Avvia infrastruttura data layer prima
docker compose up -d postgres postgres-wb redis minio
sleep 15  # attendi healthcheck

# 4. Esegui migrations
docker compose run --rm auth-service npx prisma migrate deploy
docker compose run --rm scheduling-service npx prisma migrate deploy
docker compose run --rm presenze-service npx prisma migrate deploy
docker compose run --rm cartella-service npx prisma migrate deploy
docker compose run --rm hr-service npx prisma migrate deploy
docker compose run --rm wb-service npx prisma migrate deploy

# 5. Crea bucket MinIO
docker compose run --rm minio mc mb local/clinical-attachments
docker compose run --rm minio mc mb local/quality-documents

# 6. Avvia tutto
docker compose up -d

# 7. Verifica
docker compose ps
docker compose logs -f gateway
