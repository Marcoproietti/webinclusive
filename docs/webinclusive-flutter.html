Strategia offline-first OFFLINE

L'app funziona completamente senza rete. La strategia è local-first: ogni dato letto/scritto passa prima dal DB locale (Drift/SQLite), poi sincronizzato in background.

Write → Drift (locale) → sync queue
Read → Drift (locale) sempre
Sync → quando connectivity = online
Conflict → server wins (last-write-wins)

class AuthRepositoryImpl implements AuthRepository {
  final ApiClient _api;
  final SecureStorage _storage;

  const AuthRepositoryImpl(this._api, this._storage);

  @override
  Future<AuthResult> login(String email, String password) async {
    final res = await _api.dio.post(
      '/auth/login',
      data: {'email': email, 'password': password},
    );
    final result = AuthResult.fromJson(res.data);

    // Salva token in flutter_secure_storage
    await _storage.setAccessToken(result.accessToken);
    await _storage.setUser(result.user);

    return result;
  }

  @override
  Future<void> logout() async {
    try {
      await _api.dio.post('/auth/logout');
    } finally {
      await _storage.clearAll();
    }
  }

  @override
  Future<UserEntity?> getCurrentUser() async {
    return _storage.getUser();
  }
}
core/api/auth_interceptor.dart — Auto-refresh JWT 
/// Intercetta automaticamente 401 e rinnova il token
class AuthInterceptor extends Interceptor {
  final SecureStorage _storage;
  final Dio _dio;
  bool _isRefreshing = false;

  AuthInterceptor(this._storage, this._dio);

  @override
  void onRequest(RequestOptions options,
                 RequestInterceptorHandler handler) async {
    final token = await _storage.getAccessToken();
    if (token != null) {
      options.headers['Authorization'] = 'Bearer $token';
    }
    handler.next(options);
  }

  @override
  void onError(DioException err,
               ErrorInterceptorHandler handler) async {
    if (err.response?.statusCode == 401 && !_isRefreshing) {
      _isRefreshing = true;
      try {
        // Prova a fare refresh (cookie inviato automaticamente)
        final res = await _dio.post('/auth/refresh');
        final newToken = res.data['access_token'];
        await _storage.setAccessToken(newToken);

        // Ripete la richiesta originale con nuovo token
        err.requestOptions.headers['Authorization'] =
            'Bearer $newToken';
        final retryRes = await _dio.fetch(err.requestOptions);
        handler.resolve(retryRes);
      } catch (_) {
        // Refresh fallito → logout
        await _storage.clearAll();
        handler.reject(err);
      } finally {
        _isRefreshing = false;
      }
    } else {
      handler.next(err);
    }
  }
}

import 'dart:convert';
import 'package:crypto/crypto.dart';

class CryptoService {
  final SecureStorage _storage;
  CryptoService(this._storage);

  /// Genera firma HMAC-SHA256 per check-in/out
  /// Il device_secret è salvato nel Keystore Android / Secure Enclave iOS
  Future<String> signPresence({
    required String appointmentId,
    required String operatorId,
    required double lat,
    required double lng,
    required DateTime timestamp,
  }) async {
    final deviceSecret = await _storage.getDeviceSecret();
    if (deviceSecret == null) {
      throw Exception('Device non registrato: manca device_secret');
    }

    // Payload deterministico: ordinamento fisso dei campi
    final payload = [
      appointmentId,
      operatorId,
      lat.toStringAsFixed(6),
      lng.toStringAsFixed(6),
      timestamp.toUtc().toIso8601String(),
    ].join('|');

    final key  = utf8.encode(deviceSecret);
    final msg  = utf8.encode(payload);
    final hmac = Hmac(sha256, key);
    return hmac.convert(msg).toString();
  }

  /// Registra il device secret al primo login (ricevuto dall'onboarding API)
  Future<void> registerDevice(String secret) async {
    await _storage.setDeviceSecret(secret);
  }
}
features/attendance/presentation/attendance_provider.dart 
class AttendanceNotifier
    extends AutoDisposeAsyncNotifier<AttendanceState> {

  @override
  Future<AttendanceState> build() async {
    return const AttendanceState.idle();
  }

  Future<void> checkIn(String appointmentId) async {
    state = const AsyncLoading();

    try {
      // 1. Acquisisci posizione GPS
      final loc = await ref.read(locationServiceProvider).getCurrentPosition();

      // 2. Costruisci firma HMAC
      final now = DateTime.now().toUtc();
      final user = ref.read(authProvider).valueOrNull!;
      final sig = await ref.read(cryptoServiceProvider).signPresence(
        appointmentId: appointmentId,
        operatorId: user.id,
        lat: loc.latitude, lng: loc.longitude,
        timestamp: now,
      );

      // 3. Tenta invio online
      final isOnline = ref.read(connectivityProvider).valueOrNull
                      == ConnStatus.online;

      if (isOnline) {
        final result = await ref.read(attendanceRepoProvider)
          .checkIn(CheckInRequest(
            appointmentId: appointmentId,
            lat: loc.latitude, lng: loc.longitude,
            signature: sig,
            clientTimestamp: now,
          ));
        state = AsyncData(AttendanceState.checkedIn(result));
      } else {
        // 4. Offline: accoda in SQLite
        await ref.read(offlineQueueProvider.notifier).enqueue(
          OfflineRecord.checkIn(
            appointmentId: appointmentId,
            lat: loc.latitude, lng: loc.longitude,
            signature: sig, timestamp: now,
          ),
        );
        state = AsyncData(AttendanceState.offlineQueued());
      }
    } catch (e, st) {
      state = AsyncError(e, st);
    }
  }

  Future<void> checkOut(String attendanceId) async {
    // logica analoga al checkIn
  }
}

final attendanceProvider =
  AsyncNotifierProvider.autoDispose<AttendanceNotifier, AttendanceState>(

core/services/sync_service.dart — Background Sync Worker 
import 'package:workmanager/workmanager.dart';

// Callback eseguita da Workmanager in background
@pragma('vm:entry-point')
void callbackDispatcher() {
  Workmanager().executeTask((task, data) async {
    if (task == 'syncOfflineQueue') {
      await SyncService.runSync();
    }
    return Future.value(true);
  });
}

class SyncService {
  /// Registra task periodico (ogni 15 min, min Android consentito)
  static Future<void> initialize() async {
    await Workmanager().initialize(callbackDispatcher);
    await Workmanager().registerPeriodicTask(
      'sync-offline-queue',
      'syncOfflineQueue',
      frequency: const Duration(minutes: 15),
      constraints: Constraints(
        networkType: NetworkType.connected,
      ),
    );
  }

  /// Logica di sync effettiva (chiamata anche in foreground al ripristino rete)
  static Future<void> runSync() async {
    final db  = AppDatabase();
    final api = ApiClient();

    // Recupera tutti i record non sincronizzati, ordinati per createdAt
    final pending = await db.syncQueueDao.getPending(limit: 50);
    if (pending.isEmpty) return;

    // Invia in batch all'endpoint /attendance/sync
    try {
      final res = await api.dio.post('/attendance/sync', data: {
        'records': pending.map((r) => r.payloadJson).toList(),
      });

      final processed = res.data['processed'] as int;
      final errors    = res.data['errors'] as List;

      // Marca come sincronizzati i record OK
      final errorIndices = errors.map((e) => e['index'] as int).toSet();
      for (var i = 0; i < pending.length; i++) {
        if (!errorIndices.contains(i)) {
          await db.syncQueueDao.markSynced(pending[i].id);
        } else {
          await db.syncQueueDao.incrementAttempts(pending[i].id,
            error: errors.firstWhere((e) => e['index'] == i)['reason'],
          );
        }
      }
    } catch (e) {
      // Fallimento rete: riproverà al prossimo ciclo
    }
  }

  /// Triggera sync immediata quando la rete torna online
  static void triggerOnConnectivity(WidgetRef ref) {
    ref.listen(connectivityProvider, (prev, next) {
      if (next.valueOrNull == ConnStatus.online &&
          prev?.valueOrNull == ConnStatus.offline) {
        runSync(); // fire and forget
      }
    });
  }
}

class ShiftsNotifier
    extends AutoDisposeAsyncNotifier<List<Shift>> {

  @override
  Future<List<Shift>> build() async {
    // Prima legge dal DB locale (immediato)
    final cached = await ref.read(appDatabaseProvider)
                      .shiftsDao.getThisMonth();

    // Poi aggiorna in background dal server
    _fetchFromServer();

    return cached.map(Shift.fromLocal).toList();
  }

  Future<void> _fetchFromServer() async {
    final now = DateTime.now();
    try {
      final remote = await ref.read(shiftsRepoProvider)
        .getShifts(from: DateTime(now.year, now.month, 1),
                   to:   DateTime(now.year, now.month + 1, 0));

      // Aggiorna DB locale
      await ref.read(appDatabaseProvider).shiftsDao.upsertAll(remote);
      state = AsyncData(remote);
    } catch (_) {
      // Rete non disponibile: mantieni dati locali
    }
  }
}
